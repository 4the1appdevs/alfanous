 78 -def stem(w): 

 79      """Uses the Porter stemming algorithm to remove suffixes from English 
 80      words. 
 81       
 82      >>> stem("fundamentally") 
 83      "fundament" 
 84      """ 
 85       
 86      if len(w) < 3: return w 
 87       
 88      first_is_y = w[0] == "y" 
 89      if first_is_y: 
 90          w = "Y" + w[1:] 
 91           
 92      # Step 1a 
 93      if w.endswith("s"): 
 94          if w.endswith("sses"): 
 95              w = w[:-2] 
 96          elif w.endswith("ies"): 
 97              w = w[:-2] 
 98          elif w[-2] != "s": 
 99              w = w[:-1] 
100       
101      # Step 1b 
102       
103      if w.endswith("eed"): 
104          s = w[:-3] 
105          if _mgr0.match(s): 
106              w = w[:-1] 
107      else: 
108          m = _ed_ing.match(w) 
109          if m:

111              if _s_v.match(stem): 
112                  w = stem 
113                  if _at_bl_iz.match(w): 
114                      w += "e" 
115                  elif _step1b.match(w): 
116                      w = w[:-1] 
117                  elif _c_v.match(w): 
118                      w += "e" 
119               
120      # Step 1c 
121       
122      if w.endswith("y"): 
123          stem = w[:-1] 
124          if _s_v.match(stem): 
125              w = stem + "i" 
126

128       
129      m = _step2.match(w) 
130      if m: 
131          stem = m.group(1) 
132          suffix = m.group(2)

146       
147      m = _step4_1.match(w)
